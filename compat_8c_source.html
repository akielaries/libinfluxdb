88void ic_influx_database(
89 char *host,
90 long port,
91 char *db) /* note: converts influxdb hostname to ip address */
92{
93 struct hostent *he;
94 char errorbuf[1024 + 1];
95
96 influx_port = port;
97 strncpy(influx_database, db, 256);
98
99 if (host[0] <= '0' && host[0] <= '9') {
100 DEBUG fprintf(stderr,
101 "ic_influx(ipaddr=%s,port=%ld,database=%s))\n",
102 host,
103 port,
104 db);
105 strncpy(influx_ip, host, 16);
106 } else {
107 DEBUG fprintf(stderr,
108 "ic_influx_by_hostname(host=%s,port=%ld,database=%s))\n",
109 host,
110 port,
111 db);
112 strncpy(influx_hostname, host, 1024);
113 if (isalpha(host[0])) {
114
115 he = gethostbyname(host);
116 if (he == NULL) {
117 sprintf(errorbuf,
118 "influx host=%s to ip address convertion failed "
119 "gethostbyname(), bailing out\n",
120 host);
121 error(errorbuf);
122 }
123 /* this could return multiple ip addresses but we assume its the
124 * first one */
125 if (he->h_addr_list[0] != NULL) {
126 strcpy(influx_ip, inet_ntoa(*(struct in_addr *)(he->h_addr_list[0])));
127 DEBUG fprintf(stderr,
128 "ic_influx_by_hostname hostname=%s converted to "
129 "ip address %s))\n",
130 host,
131 influx_ip);
132 } else {
133 sprintf(errorbuf,
134 "influx host=%s to ip address convertion failed (empty "
135 "list), bailing out\n",
136 host);
137 error(errorbuf);
138 }
139 } else {
140 strcpy(influx_ip,
141 host); /* perhaps the hostname is actually an ip address */
142 }
143 }
144}

155int create_socket() /* returns 1 for error and 0 for ok */
156{
157 int i;
158 static char buffer[4096];
159 static struct sockaddr_in serv_addr;
160
161 if (debug)
162 DEBUG fprintf(stderr,
163 "socket: trying to connect to \"%s\":%ld\n",
164 influx_ip,
165 influx_port);
166 if ((sockfd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
167 error("socket() call failed");
168 return 0;
169 }
170
171 serv_addr.sin_family = AF_INET;
172 serv_addr.sin_addr.s_addr = inet_addr(influx_ip);
173 serv_addr.sin_port = htons(influx_port);
174
175 /* connect tot he socket offered by the web server */
176 if (connect(sockfd, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {
177 DEBUG fprintf(stderr, " connect() call failed errno=%d", errno);
178 return 0;
179 }
180 return 1;
181}

227void ic_sub(char *resource) {
228 int i;
229
230 ic_check(strlen(saved_section) + strlen(influx_tags) + strlen(saved_sub) +
231 strlen(resource) + 9);
232
233 /* remove previously added section */
234 if (first_sub) {
235 for (i = output_char - 1; i > 0; i--) {
236 if (output[i] == '\n') {
237 output[i + 1] = 0;
238 output_char = i + 1;
239 break;
240 }
241 }
242 }
243 first_sub = 0;
244
245 /* remove the trailing s */
246 strcpy(saved_sub, saved_section);
247 if (saved_sub[strlen(saved_sub) - 1] == 's') {
248 saved_sub[strlen(saved_sub) - 1] = 0;
249 }
250 output_char += sprintf(&output[output_char],
251 "%s,%s,%s_name=%s ",
252 saved_section,
253 influx_tags,
254 saved_sub,
255 resource);
256 subended = 0;
257 DEBUG fprintf(stderr, "ic_sub(\"%s\") count=%ld\n", resource, output_char);
258}

309void ic_push() {
310 char header[1024];
311 char result[1024];
312 char buffer[1024 * 8];
313 int ret;
314 int i;
315 int total;
316 int sent;
317 int code;
318
319 if (output_char == 0) /* nothing to send so skip this operation */
320 return;
321 if (influx_port) {
322 DEBUG fprintf(stderr, "ic_push() size=%ld\n", output_char);
323 if (create_socket() == 1) {
324
325 sprintf(buffer,
326 "POST /write?db=%s&u=%s&p=%s HTTP/1.1\r\nHost: "
327 "%s:%ld\r\nContent-Length: %ld\r\n\r\n",
328 influx_database,
329 influx_username,
330 influx_password,
331 influx_hostname,
332 influx_port,
333 output_char);
334 DEBUG fprintf(stderr,
335 "buffer size=%ld\nbuffer=<%s>\n",
336 strlen(buffer),
337 buffer);
338 if ((ret = write(sockfd, buffer, strlen(buffer))) != strlen(buffer)) {
339 fprintf(stderr,
340 "warning: \"write post to sockfd failed.\" errno=%d\n",
341 errno);
342 }
343 total = output_char;
344 sent = 0;
345 if (debug == 2)
346 fprintf(stderr, "output size=%d output=\n<%s>\n", total, output);
347 while (sent < total) {
348 ret = write(sockfd, &output[sent], total - sent);
349 DEBUG fprintf(stderr,
350 "written=%d bytes sent=%d total=%d\n",
351 ret,
352 sent,
353 total);
354 if (ret < 0) {
355 fprintf(stderr,
356 "warning: \"write body to sockfd failed.\" errno=%d\n",
357 errno);
358 break;
359 }
360 sent = sent + ret;
361 }
362 for (i = 0; i < 1024; i++) /* empty the buffer */
363 result[i] = 0;
364 if ((ret = read(sockfd, result, sizeof(result))) > 0) {
365 result[ret] = 0;
366 DEBUG fprintf(stderr, "received bytes=%d data=<%s>\n", ret, result);
367 sscanf(result, "HTTP/1.1 %d", &code);
368 for (i = 13; i < 1024; i++)
369 if (result[i] == '\r')
370 result[i] = 0;
371 if (debug == 2)
372 fprintf(stderr,
373 "http-code=%d text=%s [204=Success]\n",
374 code,
375 &result[13]);
376 if (code != 204)
377 fprintf(stderr, "code %d -->%s<--\n", code, result);
378 }
379 close(sockfd);
380 sockfd = 0;
381 DEBUG fprintf(stderr, "ic_push complete\n");
382 } else {
383 DEBUG fprintf(stderr, "socket create failed\n");
384 }
385 } else
386 error("influx port is not set, bailing out");
387
388 output[0] = 0;
389 output_char = 0;
390}
