31 perror("Invalid address or address not supported");

37 if (connect(sockfd, (struct sockaddr *)&server_addr, sizeof(server_addr)) <

42 }

47// helper function to send an HTTP request
48int send_http_request(int sockfd, const char *request) {
49 printf("[+] Sending request: %s\n\n", request);
50
51 size_t total_sent = 0;
52 size_t request_len = strlen(request);
53
54 // send the HTTP request
55 while (total_sent < request_len) {
56 ssize_t sent = send(sockfd, request + total_sent, request_len - total_sent, 0);
57 if (sent < 0) {
58 perror("Send failed");
59 return -1;
60 }
61 total_sent += sent;
62 }
63
64 // buffer to store the HTTP response
65 char response[MAX_BUF_SIZE] = {0};
66 ssize_t received = recv(sockfd, response, sizeof(response) - 1, 0);
67 if (received < 0) {
68 perror("Failed to receive response");
69 return -1;
70 }
71
72 // print the received response
73 printf("[+] Received response:\n%s\n", response);
74
75 // check for HTTP 2xx status code to confirm success
76 if (strstr(response, "HTTP/1.1 2") == NULL) {
77 fprintf(stderr, "Request failed with response: %s\n", response);
78 return -1;
79 }
80
81 return 0;
82}
